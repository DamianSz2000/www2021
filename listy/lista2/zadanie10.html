let inp = document.getElementsByTagName("input"); //przypisujemy zmiennej inp wartości wszystkich pól tekstowych input

let a = +inp[0].value, b = +inp[1].value, c = +inp[2].value; //przypisujemy zmiennej a, b, c kolejne indeksy zmiennej inp która jest teraz tablicą złożoną z wartości pól input
let delta = b * b - 4 * a * c; //przypisujemy zmiennej delta wynik obliczenia delty na podstawie podanych zmiennych
if (delta < 0) { //jeżeli delta mniejsza niż 0
document.getElementsByTagName("div")[0].innerHTML = "Brak rozwiązań"; //wstawiamy do pierwszego tagu div dokumentu komunikat o braku rozwiązań
} else if (delta == 0) { //jeśli delta to 0
let x0 = -b / (2 * a); //przypisujemy zmiennej x0 jedyny pierwiastek równania
document.getElementsByTagName("div")[0].innerHTML = "x0=" + x0; //i wstawiamy go do pierwszego tagu div dokumentu
} else { //w przypadku w którym żadna z powyższych warunków nie jest spełniony
let x1 = (-b - Math.sqrt(delta)) / (2 * a); //przypisujemy zmiennej x1 pierwszy pierwiastek równania
let x2 = (-b + Math.sqrt(delta)) / (2 * a); //przypisujemy zmiennej x2 drugi pierwiastek równania
document.getElementsByTagName("div")[0].innerHTML = "x1=" + x1 + "x2=" + x2; //I wstawiamy wynik do pierwszego tagu div dokument
}
